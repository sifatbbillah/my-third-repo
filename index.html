//nice /////////// Original array const numbers = [1, 2, 3, 4, 5, 6]; // 1. map() → double each number const doubled = numbers.map(num => num * 2); console.log("Mapped (doubled):", doubled); // 2. filter() → keep only even numbers const evens = numbers.filter(num => num % 2 === 0); console.log("Filtered (evens):", evens); // 3. find() → find the first number greater than 3 const found = numbers.find(num => num > 3); console.log("Found first > 3:", found); // 4. forEach() → loop and print each number console.log("forEach print:"); numbers.forEach(num => { console.log(num); });
